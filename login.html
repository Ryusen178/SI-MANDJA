async function login() { const email = document.getElementById("email").value; const password = document.getElementById("password").value; const { data, error } = await supa.auth.signInWithPassword({ email, password }); if (error) { alert("Login gagal: " + error.message); } else { // Ambil role user dari tabel "users" const { data: userData } = await supa .from("users") .select("role") .eq("email", email) .single(); // Simpan role ke localStorage localStorage.setItem("role", userData.role); // Redirect ke dashboard window.location.href = "dashboard.html"; } }
